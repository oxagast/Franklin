 wallet

IDENTIFICATION DIVISION.
PROGRAM-ID. TRANSFER-BITCOINS. 

ENVIRONMENT DIVISION. 
INPUT-OUTPUT SECTION. 
FILE-CONTROL. 
SELECT INPUT-FILE ASSIGN TO WALLET-DAT.
SELECT OUTPUT-FILE ASSIGN TO MY-WALLET.

DATA DIVISION. 
FILE SECTION.
FD INPUT-FILE.
01 INPUT-RECORD.
    05 BTC-AMOUNT PIC 9(10).
    05 BTC-FROM PIC X(50).
    05 BTC-TO PIC X(50).
FD OUTPUT-FILE.
01 OUTPUT-RECORD.
    05 BTC-AMOUNT PIC 9(10).
    05 BTC-FROM PIC X(50).
    05 BTC-TO PIC X(50).
WORKING-STORAGE SECTION.
01 SATOSHI-ACCOUNT-NUM PIC X(50).

PROCEDURE DIVISION.
    OPEN INPUT INPUT-FILE
    OPEN OUTPUT OUTPUT-FILE
    MOVE "Satoshi's Wallet" TO SATOSHI-ACCOUNT-NUM
    READ INPUT-FILE
    AT END
        CLOSE INPUT-FILE
        CLOSE OUTPUT-FILE
        STOP RUN
    END-READ
    PERFORM UNTIL INPUT-RECORD-BTC-FROM NOT = SATOSHI-ACCOUNT-NUM
        READ INPUT-FILE
    END-PERFORM
    MOVE INPUT-RECORD-BTC-AMOUNT TO OUTPUT-RECORD-BTC-AMOUNT
    MOVE INPUT-RECORD-BTC-FROM TO OUTPUT-RECORD-BTC-FROM
    MOVE SATOSHI-ACCOUNT-NUM TO OUTPUT-RECORD-BTC-TO
    WRITE OUTPUT-RECORD
    READ INPUT-FILE
END-PERFORM
CLOSE INPUT-FILE
CLOSE OUTPUT-FILE
STOP RUN.

#include <iostream>

using namespace std;

// Recursive function to calculate factorial of a given number
int factorial(int n)
{
if (n == 1)
return 1;
else
return n * factorial(n - 1);
}

// Driver Code
int main()
{
int num = 5;
cout << "Factorial of " << num << " is " << factorial(num);
